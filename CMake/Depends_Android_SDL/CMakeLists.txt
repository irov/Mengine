CMAKE_MINIMUM_REQUIRED( VERSION 2.6 )

INCLUDE( ../android_template.cmake )

PROJECT( Depends_Android_SDL )

SET(CMAKE_SYSTEM_NAME Android)

SET(MENGINE_RENDER_OPENGL ON CACHE BOOL "MENGINE_RENDER_OPENGL" FORCE)
SET(MENGINE_RENDER_DIRECTX OFF CACHE BOOL "MENGINE_RENDER_DIRECTX" FORCE)

SET(MENGINE_PLATFORM_WIN32 OFF CACHE BOOL "MSVC" FORCE)
SET(MENGINE_PLATFORM_ANDROID ON CACHE BOOL "MSVC" FORCE)

SET(MENGINE_TARGET_SDL ON CACHE BOOL "SDL" FORCE)
SET(MENGINE_TARGET_GRADLE ON CACHE BOOL "MSVC" FORCE)
SET(MENGINE_TARGET_MSVC OFF CACHE BOOL "MSVC" FORCE)
SET(MENGINE_TARGET_MINGW OFF CACHE BOOL "MENGINE_TARGET_MINGW" FORCE)

SET(MENGINE_PLUGIN_SPINE OFF CACHE BOOL "Spine" FORCE)
SET(MENGINE_PLUGIN_TTF ON CACHE BOOL "TTF" FORCE)
SET(MENGINE_PLUGIN_CURL ON CACHE BOOL "CURL" FORCE)

SET( MENGINE_REPOSITORY ${CMAKE_CURRENT_SOURCE_DIR}/../.. )
SET( THIRDPARTY_DIR ${MENGINE_REPOSITORY}/dependencies )

set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})

SET( CMAKE_TEMP_DIR ${MENGINE_REPOSITORY}/CMakeTemp/Depends_Android/${CMAKE_GENERATOR}/${CMAKE_BUILD_TYPE} )

ADD_SUBDIRECTORY( ${CMAKE_CURRENT_SOURCE_DIR}/../Dependencies ${CMAKE_TEMP_DIR} )
