 
CMAKE_MINIMUM_REQUIRED( VERSION 2.6 )

MACRO( ADD_FILTER group_name )
	SOURCE_GROUP( ${group_name} FILES ${ARGN} )
	SET( SRC_FILES ${SRC_FILES} ${ARGN} )
	#MESSAGE( ${ARGN} )
ENDMACRO( ADD_FILTER group_name files )

# solution
PROJECT( Mengine_MACOS )

SET(MENGINE_TARGET "MACOS")

INCLUDE( ../macos_template.cmake )

SET( MENGINE_REPOSITORY ${CMAKE_CURRENT_SOURCE_DIR}/../.. )
SET( MENGINE_LIB_PREFIX )
SET( MENGINE_LIB_SUFFIX )


# output paths
SET( CMAKE_RUNTIME_OUTPUT_DIRECTORY ${MENGINE_REPOSITORY}/bin/MacOS )

SET( CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR} )

# input and thirdparty paths
SET( THIRDPARTY_DIR ${MENGINE_REPOSITORY}/dependencies )
SET( SOURCE_DIR ${MENGINE_REPOSITORY}/src )
SET( CMAKE_TEMP_DIR ${MENGINE_REPOSITORY}/CMakeTemp/MACOS${CMAKE_BUILD_TYPE} )

# implementation

SET( MENGE_DEBUG )
string(COMPARE EQUAL CMAKE_BUILD_TYPE Debug MENGE_DEBUG)

SET( MENGINE_LIB_PREFIX "lib" )
SET( MENGINE_LIB_SUFFIX ".a" )

SET( MENGINE_LIB_POSTFIX ${MENGINE_LIB_SUFFIX} )
SET( MENGINE_LIB_SUFFIX ${CMAKE_DEBUG_POSTFIX}${MENGINE_LIB_SUFFIX} )

ADD_DEFINITIONS(-D"override = /* */")
ADD_DEFINITIONS(-D"MENGINE_CONFIG=Config_MACOS.h")
ADD_DEFINITIONS(-DMENGE_MASTER_RELEASE)

configure_file(${CMAKE_CURRENT_SOURCE_DIR}/Config_MACOS.h ${SOURCE_DIR}/Config/Config_MACOS.h COPYONLY)

SET( MENGE_DEPENDENCIES )
SET( APPLICATION_DEPENDENCIES )
SET( APPLICATION_LIBS )
SET( APPLICATION_LIB_PATHS )

INCLUDE_DIRECTORIES( ${SOURCE_DIR} )
INCLUDE_DIRECTORIES( ${SOURCE_DIR}/Utils )
INCLUDE_DIRECTORIES( ${THIRDPARTY_DIR}/boost )

ADD_SUBDIRECTORY( ${SOURCE_DIR} ${CMAKE_TEMP_DIR}/src )

LINK_DIRECTORIES( ${MENGINE_REPOSITORY}/xcode_depends_macos ${APPLICATION_LIB_PATHS} )

ADD_SUBDIRECTORY( ${SOURCE_DIR}/MacOSApplication ${CMAKE_TEMP_DIR}/MacOSApplication )

ADD_DEPENDENCIES( MacOSApplication ${APPLICATION_DEPENDENCIES} )
