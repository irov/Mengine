CMAKE_MINIMUM_REQUIRED(VERSION 3.10)

OPTION(MENGINE_USE_PRECOMPILED_HEADER "Mengine use precompiled header" OFF)

PROJECT(SDL_Android_SDL)

SET(MENGINE_REPOSITORY ${CMAKE_CURRENT_SOURCE_DIR}/../../..)

INCLUDE(${MENGINE_REPOSITORY}/cmake/base_template.cmake)
INCLUDE(${MENGINE_REPOSITORY}/cmake/macro_template.cmake)
INCLUDE(${MENGINE_REPOSITORY}/cmake/android_template.cmake)
INCLUDE(${MENGINE_REPOSITORY}/cmake/dependencies_template.cmake)

SET_MENGINE_ENVIRONMENT(ANDROID OPENGL SDL GRADLE)

SET(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${THIRDPARTY_LIB_OUT}/${CMAKE_BUILD_TYPE}/${ANDROID_ABI})
SET(CMAKE_TEMP_DIR ${CMAKE_BINARY_DIR}/Dependencies/${ANDROID_ABI})

MESSAGE("CMAKE_ARCHIVE_OUTPUT_DIRECTORY: ${CMAKE_ARCHIVE_OUTPUT_DIRECTORY}")

SET(MENGINE_ENABLE_DEPENDENCE_ONLY "MENGINE_DEPENDENCE_SDL" CACHE STRING "MENGINE_ENABLE_DEPENDENCE_ONLY" FORCE)

ADD_SUBDIRECTORY(${CMAKE_CURRENT_SOURCE_DIR}/../../Dependencies ${CMAKE_TEMP_DIR})
