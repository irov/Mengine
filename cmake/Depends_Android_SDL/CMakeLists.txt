CMAKE_MINIMUM_REQUIRED(VERSION 3.10)

OPTION(MENGINE_USE_PRECOMPILED_HEADER "Mengine use precompiled header" OFF)

INCLUDE(${CMAKE_CURRENT_SOURCE_DIR}/../base_template.cmake)
