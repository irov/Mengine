PROJECT(Depends_WIN32_UWP)

SET(MENGINE_REPOSITORY ${CMAKE_CURRENT_SOURCE_DIR}/../..)

INCLUDE(${MENGINE_REPOSITORY}/cmake/base_template.cmake)

SET_MENGINE_ENVIRONMENT(UWP DIRECTX11 SDL MSVC)

SET(MENGINE_DEPENDENCE_CURL OFF CACHE BOOL "MENGINE_DEPENDENCE_CURL" FORCE)

SET(MENGINE_DEPENDENCE_PYTHON3 OFF CACHE BOOL "MENGINE_DEPENDENCE_PYTHON3" FORCE)
SET(MENGINE_DEPENDENCE_PYBIND3 OFF CACHE BOOL "MENGINE_DEPENDENCE_PYBIND3" FORCE)

ADD_SUBDIRECTORY(${CMAKE_CURRENT_SOURCE_DIR}/../Dependencies ${CMAKE_TEMP_DIR})
