cmake_minimum_required(VERSION 3.16)

project(Arena3D_WIN64_DX11)

set(PROJECT_DIR ${CMAKE_CURRENT_SOURCE_DIR}/../..)
set(PROJECT_TEMP_DIR ${CMAKE_BINARY_DIR}/solutions)
set(MENGINE_CMAKE_DIR ${PROJECT_DIR}/../../cmake/Win64_DX11)

# Resource discovery happens at configure time on the application targets.
find_package(Python3 REQUIRED COMPONENTS Interpreter)
execute_process(
    COMMAND ${Python3_EXECUTABLE} ${PROJECT_DIR}/tools/arena3d_content.py build
    WORKING_DIRECTORY ${PROJECT_DIR}
    RESULT_VARIABLE ARENA3D_CONTENT_BUILD_RESULT
)
if(NOT ARENA3D_CONTENT_BUILD_RESULT EQUAL 0)
    message(FATAL_ERROR "Arena3D content generation failed: ${ARENA3D_CONTENT_BUILD_RESULT}")
endif()
execute_process(
    COMMAND ${Python3_EXECUTABLE} ${PROJECT_DIR}/tools/arena3d_content.py check
    WORKING_DIRECTORY ${PROJECT_DIR}
    RESULT_VARIABLE ARENA3D_CONTENT_CHECK_RESULT
)
if(NOT ARENA3D_CONTENT_CHECK_RESULT EQUAL 0)
    message(FATAL_ERROR "Arena3D content validation failed: ${ARENA3D_CONTENT_CHECK_RESULT}")
endif()
set(MENGINE_PROJECT_NAME "Arena3D_WIN64_DX11" CACHE STRING "Mengine project name" FORCE)
set(MENGINE_DEPLOY_PATH ${PROJECT_DIR}/resources CACHE STRING "MENGINE_DEPLOY_PATH" FORCE)

if(NOT DEFINED MENGINE_DEPENDENCIES_PROJECT OR MENGINE_DEPENDENCIES_PROJECT STREQUAL "")
    set(MENGINE_DEPENDENCIES_PROJECT "Depends_WIN64_DX11" CACHE STRING "MENGINE_DEPENDENCIES_PROJECT" FORCE)
endif()

macro(MENGINE_PREPARE_FUNCTION)
    DISABLE_PLUGIN(MENGINE_PLUGIN_PYTHONFRAMEWORK)
    ENABLE_PLUGIN(MENGINE_PLUGIN_GLTF_IMPORTER)
    DISABLE_PLUGIN(MENGINE_PLUGIN_ASTRALAX)
    DISABLE_PLUGIN(MENGINE_PLUGIN_BOX2D)
    DISABLE_PLUGIN(MENGINE_PLUGIN_MOVIE)
    DISABLE_PLUGIN(MENGINE_PLUGIN_MOVIE1)
    DISABLE_PLUGIN(MENGINE_PLUGIN_VIDEO)
    DISABLE_PLUGIN(MENGINE_PLUGIN_THEORA)
endmacro()

set(MENGINE_BUILD_MENGINE_SCRIPT_EMBEDDED OFF CACHE BOOL "MENGINE_BUILD_MENGINE_SCRIPT_EMBEDDED" FORCE)
set(MENGINE_EXTERNAL_PROJECT_NAME "Arena3DGame" CACHE STRING "MENGINE_EXTERNAL_PROJECT_NAME" FORCE)
set(MENGINE_EXTERNAL_SOURCE ON CACHE BOOL "MENGINE_EXTERNAL_SOURCE" FORCE)
set(MENGINE_EXTERNAL_SOURCE_PATH ${PROJECT_DIR}/src CACHE STRING "MENGINE_EXTERNAL_SOURCE_PATH" FORCE)
set(MENGINE_EXTERNAL_TEMP_PATH ${PROJECT_TEMP_DIR}/Arena3DGame CACHE STRING "MENGINE_EXTERNAL_TEMP_PATH" FORCE)
set(MENGINE_USE_SUBFOLDER ON CACHE STRING "MENGINE_USE_SUBFOLDER" FORCE)
set(MENGINE_SUBFOLDER_NAME "Mengine" CACHE STRING "MENGINE_SUBFOLDER_NAME" FORCE)

add_subdirectory(${MENGINE_CMAKE_DIR} ${PROJECT_TEMP_DIR}/Mengine)

if(MSVC)
    set_property(TARGET Win32Application PROPERTY VS_DEBUGGER_WORKING_DIRECTORY "${PROJECT_DIR}/resources/Data")
endif()
