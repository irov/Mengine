CMAKE_MINIMUM_REQUIRED(VERSION 3.2)

PROJECT(Palette_WIN32)

SET(PROJECT_DIR ${CMAKE_CURRENT_SOURCE_DIR}/../..)
SET(PROJECT_TEMP_DIR ${CMAKE_BINARY_DIR}/solutions)

#path to Mengine cmake
SET(MENGINE_CMAKE_DIR ${PROJECT_DIR}/../../CMake/Win32)

set(MENGINE_USE_PYTHON_FRAMEWORK OFF CACHE BOOL "MENGINE_USE_PYTHON_FRAMEWORK" FORCE)
set(MENGINE_EXTERNAL_FRAMEWORK ON CACHE BOOL "MENGINE_EXTERNAL_FRAMEWORK" FORCE)
set(MENGINE_EXTERNAL_FRAMEWORK_SHARED OFF CACHE BOOL "MENGINE_EXTERNAL_FRAMEWORK_SHARED" FORCE)
set(MENGINE_EXTERNAL_FRAMEWORK_NAME "PaletteFramework" CACHE STRING "MENGINE_EXTERNAL_FRAMEWORK_NAME" FORCE)
set(MENGINE_EXTERNAL_FRAMEWORK_SOURCE_PATH ${PROJECT_DIR}/src CACHE STRING "MENGINE_EXTERNAL_FRAMEWORK_SOURCE_PATH" FORCE)
set(MENGINE_EXTERNAL_FRAMEWORK_TEMP_PATH ${PROJECT_TEMP_DIR}/PaletteFramework CACHE STRING "MENGINE_EXTERNAL_FRAMEWORK_TEMP_PATH" FORCE)
set(MENGINE_USE_SUBFOLDER ON CACHE STRING "MENGINE_USE_SUBFOLDER" FORCE)
set(MENGINE_SUBFOLDER_NAME "Mengine" CACHE STRING "MENGINE_SUBFOLDER_NAME" FORCE)

ADD_SUBDIRECTORY(${MENGINE_CMAKE_DIR} ${PROJECT_TEMP_DIR}/Mengine)
