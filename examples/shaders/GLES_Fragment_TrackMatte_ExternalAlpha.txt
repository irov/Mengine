uniform sampler2D inSampler0;
uniform sampler2D inSampler1;
uniform sampler2D inSampler2;

uniform highp vec4 uvbb;

varying highp vec2 v_UV0;
varying highp vec2 v_UV1;
varying mediump vec4 v_Col;

void main(void)
{
    highp vec2 uv1_clamp;
    uv1_clamp.x = clamp(v_UV1.x, uvbb.x, uvbb.z);
    uv1_clamp.y = clamp(v_UV1.y, uvbb.y, uvbb.w);

	mediump vec4 c0 = texture2D(inSampler0, v_UV0);
    mediump vec4 a0 = texture2D(inSampler1, v_UV0);
    mediump vec4 c1 = texture2D(inSampler2, uv1_clamp);
    
	mediump vec4 c = v_Col * c0;
    c.a *= a0.a;
    c.a *= c1.a;
    
	gl_FragColor = c;
}
