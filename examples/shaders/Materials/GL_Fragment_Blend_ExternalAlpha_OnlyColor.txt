#version 330 core

uniform sampler2D inSampler0;
uniform sampler2D inSampler1;

in vec2 v_UV0;
in vec2 v_UV1;
in vec4 v_Col;

out vec4 fragColor;

void main(void)
{
    vec4 c0 = texture(inSampler0, v_UV0);
    vec4 c1 = texture(inSampler1, v_UV1);
    vec4 c = v_Col;
    c.w = v_Col.w * c0.w * c1.w;
    
    fragColor = c;
}
