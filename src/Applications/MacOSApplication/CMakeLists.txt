if(MENGINE_APPLICATION_DEPLOY_DATA)

        MESSAGE(FATAL_ERROR "Deploy data path setup invalid path '${MENGINE_DEPLOY_PATH}/${MENGINE_APPLICATION_DEPLOY_DATA}'")
    endif()

    file(GLOB_RECURSE FILES ${MENGINE_DEPLOY_PATH}/${MENGINE_APPLICATION_DEPLOY_DATA}/*)
    list(FILTER FILES EXCLUDE REGEX ".*.DS_Store$")
    foreach(FILE ${FILES})
        file(RELATIVE_PATH RES ${MENGINE_DEPLOY_PATH}/${MENGINE_APPLICATION_DEPLOY_DATA} ${FILE})
        get_filename_component(RES_LOC ${RES} PATH)

        set_source_files_properties(${FILE} PROPERTIES MACOSX_PACKAGE_LOCATION "Resources/Data/${RES_LOC}" HEADER_FILE_ONLY TRUE)
        string(REPLACE "/" "\\" ide_group_name "deploy_data/${RES_LOC}")
        source_group(${ide_group_name} FILES ${FILE})
    endforeach()

    MESSAGE("Add data: ${MENGINE_DEPLOY_PATH}/${MENGINE_APPLICATION_DEPLOY_DATA}")
    set(MENGINE_APPLE_RESOURCES ${MENGINE_APPLE_RESOURCES} ${FILES})
endif()

if(APPLICATION_MACOS_BUNDLE_DIRECTORIES)
    list(LENGTH APPLICATION_MACOS_BUNDLE_DIRECTORIES LENGTH_APPLICATION_MACOS_BUNDLE_DIRECTORIES)
    math(EXPR LENGTH_APPLICATION_MACOS_BUNDLE_DIRECTORIES "${LENGTH_APPLICATION_MACOS_BUNDLE_DIRECTORIES}-1")

    foreach(INDEX RANGE 0 ${LENGTH_APPLICATION_MACOS_BUNDLE_DIRECTORIES} 2)
        math(EXPR DESTINATION_INDEX "${INDEX}+1")

        list(GET APPLICATION_MACOS_BUNDLE_DIRECTORIES ${INDEX} MACOS_BUNDLE_SOURCE_DIRECTORY)
        list(GET APPLICATION_MACOS_BUNDLE_DIRECTORIES ${DESTINATION_INDEX} MACOS_BUNDLE_DESTINATION_DIRECTORY)

        set(MACOS_BUNDLE_SOURCE_DIRECTORY_FULL "${MENGINE_DEPLOY_PATH}/${MACOS_BUNDLE_SOURCE_DIRECTORY}")

        if(NOT IS_DIRECTORY "${MACOS_BUNDLE_SOURCE_DIRECTORY_FULL}")
            MESSAGE(FATAL_ERROR "macOS bundle directory path setup invalid path '${MACOS_BUNDLE_SOURCE_DIRECTORY_FULL}'")
        endif()

        MESSAGE("Add macos bundle directory: ${MACOS_BUNDLE_SOURCE_DIRECTORY_FULL} -> ${MACOS_BUNDLE_DESTINATION_DIRECTORY}")

        add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
            COMMAND ${CMAKE_COMMAND} -E copy_directory
                "${MACOS_BUNDLE_SOURCE_DIRECTORY_FULL}"
                "$<TARGET_FILE_DIR:${PROJECT_NAME}>/../Resources/${MACOS_BUNDLE_DESTINATION_DIRECTORY}"
        )
    endforeach()
endif()
