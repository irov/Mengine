MENGINE_PROJECT(SDL3Platform)

ADD_FILTER(
src
    SDL3PlatformService.h
    SDL3PlatformService.cpp
    SDL3PlatformSystem.h
    SDL3PlatformSystem.cpp
    SDL3DynamicLibrary.h
    SDL3DynamicLibrary.cpp
    SDL3Input.h
    SDL3Input.cpp
)

INCLUDE_DIRECTORIES(${THIRDPARTY_CONFIG_DIR}/SDL3/include/include-config-$<LOWER_CASE:$<CONFIG>>)
INCLUDE_DIRECTORIES(${THIRDPARTY_CONFIG_DIR}/SDL3/include)
INCLUDE_DIRECTORIES(${THIRDPARTY_DIR}/SDL3/include)

ADD_MENGINE_LIBRARY(Platforms)

TARGET_LINK_LIBRARIES(${PROJECT_NAME} Kernel)

IF(MENGINE_TARGET_MACOS)
    TARGET_LINK_LIBRARIES(${PROJECT_NAME} MacOS)
ENDIF()

IF(MENGINE_TARGET_APPLE)
    TARGET_LINK_LIBRARIES(${PROJECT_NAME} Apple)
ENDIF()

IF(MENGINE_TARGET_WINDOWS)
    TARGET_LINK_LIBRARIES(${PROJECT_NAME} Windows)
    TARGET_LINK_LIBRARIES(${PROJECT_NAME} Shlwapi.lib)
ENDIF()
