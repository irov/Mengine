MENGINE_PROJECT(SDLPlatform)

ADD_FILTER(
src
    SDLPlatformService.h
    SDLPlatformService.cpp
    SDLPlatformSystem.h
    SDLPlatformSystem.cpp
    SDLDynamicLibrary.h
    SDLDynamicLibrary.cpp
    SDLInput.h
    SDLInput.cpp
)

IF(MENGINE_TARGET_MACOS)
    ADD_FILTER(
    macos
        MacOSWorkspace.h
        MacOSWorkspace.mm
    )
ENDIF()

INCLUDE_DIRECTORIES(${THIRDPARTY_CONFIG_DIR}/SDL/include/include-config-$<LOWER_CASE:$<CONFIG>>)
INCLUDE_DIRECTORIES(${THIRDPARTY_CONFIG_DIR}/SDL/include)
INCLUDE_DIRECTORIES(${THIRDPARTY_DIR}/SDL/include)

ADD_MENGINE_LIBRARY(Platforms)

TARGET_LINK_LIBRARIES(${PROJECT_NAME} Kernel)

IF(MENGINE_TARGET_MACOS)
    TARGET_LINK_LIBRARIES(${PROJECT_NAME} MacOS)
ENDIF()

IF(MENGINE_TARGET_APPLE)
    TARGET_LINK_LIBRARIES(${PROJECT_NAME} Apple)
ENDIF()

IF(MENGINE_TARGET_ANDROID)
    TARGET_LINK_LIBRARIES(${PROJECT_NAME} Android)
ENDIF()

IF(MENGINE_TARGET_WINDOWS)
    TARGET_LINK_LIBRARIES(${PROJECT_NAME} Windows)
    TARGET_LINK_LIBRARIES(${PROJECT_NAME} Shlwapi.lib)
ENDIF()
