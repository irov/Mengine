src
    ImGUIInterface.h

    ImGUIPlugin.cpp
    ImGUIPlugin.h
    ImGUIModule.cpp
    ImGUIModule.h
    
    ImGUIRender.cpp
    ImGUIRender.h
)

if(MENGINE_PLATFORM_WIN32)
    ADD_FILTER(
    imgui_platform_ext
        ${THIRDPARTY_DIR}/imgui/examples/imgui_impl_win32.h
        ${THIRDPARTY_DIR}/imgui/examples/imgui_impl_win32.cpp
    )

if(MENGINE_PLATFORM_SDL)
    ADD_FILTER(
    imgui_platform_ext
        ${THIRDPARTY_DIR}/imgui/examples/imgui_impl_sdl.h
        ${THIRDPARTY_DIR}/imgui/examples/imgui_impl_sdl.cpp

    imgui_render_ext

    )

    ADD_FILTER(
    imgui_render_ext
