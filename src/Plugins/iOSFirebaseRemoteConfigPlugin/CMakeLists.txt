MENGINE_PROJECT(iOSFirebaseRemoteConfigPlugin)

IF(NOT MENGINE_APPLICATION_DEPLOY_FIREBASE_REMOTECONFIG_PLIST_PATH)
    MESSAGE(FATAL_ERROR "iOS Firebase RemoteConfig required setup MENGINE_APPLICATION_DEPLOY_FIREBASE_REMOTECONFIG_PLIST_PATH")
ENDIF()

get_filename_component(MENGINE_APPLICATION_DEPLOY_FIREBASE_REMOTECONFIG_PLIST_NAME ${MENGINE_DEPLOY_PATH}/${MENGINE_APPLICATION_DEPLOY_FIREBASE_REMOTECONFIG_PLIST_PATH} NAME_WE)

MESSAGE("iOS Firebase RemoteConfig plist path: ${MENGINE_APPLICATION_DEPLOY_FIREBASE_REMOTECONFIG_PLIST_PATH}")
MESSAGE("iOS Firebase RemoteConfig plist name: ${MENGINE_APPLICATION_DEPLOY_FIREBASE_REMOTECONFIG_PLIST_NAME}")

ADD_FILTER(
src
    iOSFirebaseRemoteConfigPlugin.h
    iOSFirebaseRemoteConfigPlugin.mm

    iOSFirebaseRemoteConfigInterface.h
)

if(MENGINE_BUILD_MENGINE_SCRIPT_EMBEDDED)
    ADD_FILTER(
    embedding
        iOSFirebaseRemoteConfigScriptEmbedding.h
        iOSFirebaseRemoteConfigScriptEmbedding.mm
        )

    INCLUDE_DIRECTORIES(${THIRDPARTY_DIR}/pybind/include)
endif()

ADD_DEFINITIONS(-DMENGINE_FIREBASE_REMOTECONFIG_PLIST_NAME="${MENGINE_APPLICATION_DEPLOY_FIREBASE_REMOTECONFIG_PLIST_NAME}")

ADD_MENGINE_PLUGIN(MENGINE_PLUGIN_IOS_FIREBASE_REMOTECONFIG)

ADD_MENGINE_COCOAPOD_GOOGLE_UTILITIES()
ADD_MENGINE_COCOAPOD("FirebaseRemoteConfig" "NO-GIT" "12.16.0")

ADD_MENGINE_APPLE_APPLICATION_DELEGATE("iOSFirebaseRemoteConfigPlugin")

ADD_MENGINE_APPLE_RESOURCES(${MENGINE_APPLICATION_DEPLOY_FIREBASE_REMOTECONFIG_PLIST_PATH})

if(MENGINE_BUILD_MENGINE_SCRIPT_EMBEDDED)
    TARGET_LINK_LIBRARIES(${PROJECT_NAME} Python)
endif()

TARGET_LINK_LIBRARIES(${PROJECT_NAME} iOSFirebasePlugin)
