MENGINE_PROJECT(iOSSentryPlugin)

ADD_FILTER(
src
    iOSSentryNotificationObserver.h
    iOSSentryNotificationObserver.mm

    iOSSentryPlugin.h
    iOSSentryPlugin.mm
)

ADD_MENGINE_PLUGIN(MENGINE_PLUGIN_IOS_SENTRY)

ADD_MENGINE_APPLE_APPLICATION_DELEGATE("iOSSentryPlugin")

TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${THIRDPARTY_LIB_DIR}/${MENGINE_LIB_PREFIX}math${MENGINE_LIB_SUFFIX})
TARGET_LINK_LIBRARIES(${PROJECT_NAME} Kernel)

ADD_MENGINE_COCOAPOD(Sentry "https://github.com/getsentry/sentry-cocoa.git" "9.19.1")

if(MENGINE_BUILD_TYPE_RELEASE)
    IF(NOT MENGINE_IOS_SENTRY_CLI_PATH)
        MESSAGE(FATAL_ERROR "Sentry upload symbols CLI required setup MENGINE_IOS_SENTRY_CLI_PATH")
    ENDIF()

    IF(NOT MENGINE_IOS_SENTRY_CLI_ORG)
        MESSAGE(FATAL_ERROR "Sentry upload symbols CLI required setup MENGINE_IOS_SENTRY_CLI_ORG")
    ENDIF()

    IF(NOT MENGINE_IOS_SENTRY_CLI_PROJECT)
        MESSAGE(FATAL_ERROR "Sentry upload symbols CLI required setup MENGINE_IOS_SENTRY_CLI_PROJECT")
    ENDIF()

    IF(NOT MENGINE_IOS_SENTRY_CLI_AUTH_TOKEN)
        MESSAGE(FATAL_ERROR "Sentry upload symbols CLI required setup MENGINE_IOS_SENTRY_CLI_AUTH_TOKEN")
    ENDIF()

    ADD_MENGINE_SCRIPT_PHASE("Upload Sentry Symbols" "export SENTRY_ORG=${MENGINE_IOS_SENTRY_CLI_ORG}\nexport SENTRY_PROJECT=${MENGINE_IOS_SENTRY_CLI_PROJECT}\nexport SENTRY_AUTH_TOKEN=${MENGINE_IOS_SENTRY_CLI_AUTH_TOKEN}\n$(${MENGINE_IOS_SENTRY_CLI_PATH} upload-dif --include-sources \"${CMAKE_RUNTIME_OUTPUT_DIRECTORY}\" 2>&1 >/dev/null)" "['($){DWARF_DSYM_FOLDER_PATH}/($){DWARF_DSYM_FILE_NAME}/Contents/Resources/DWARF/($){TARGET_NAME}']")
endif()
